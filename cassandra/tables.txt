

CREATE TABLE world_day_tweets (
    dayslot int,
    topic text,
    count int,
    PRIMARY KEY (dayslot, count)
) WITH CLUSTERING ORDER BY (count DESC)

CREATE TABLE world_hour_tweets (
    hourslot int,
    topic text,
    count int,
    PRIMARY KEY (hourslot, count)
) WITH CLUSTERING ORDER BY (count DESC)

CREATE TABLE world_minute_tweets (
    minuteslot int,
    topic text,
    count int,
    PRIMARY KEY (minuteslot, count)
) WITH CLUSTERING ORDER BY (count DESC)
